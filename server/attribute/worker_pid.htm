swoole_server::$worker_pid

得到当前Worker进程的操作系统进程ID。与posix_getpid()的返回值相同。

int $serv->worker_pid;

$serv->on('workerStart', function($serv, $worker_id) {  
    echo "WorkerStart: MasterPid={$serv->master_pid}|Manager_pid={$serv->manager_pid}|
            WorkerId={$serv->worker_id}|WorkerPid={$serv->worker_pid}\n";  
});
